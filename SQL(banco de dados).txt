CREATE TABLE IF NOT EXISTS `formulario_anamnese` (
    `id` INT(11) NOT NULL AUTO_INCREMENT,
    `data_cadastro` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    `nome_completo` VARCHAR(255) NOT NULL,
    `cpf_rg` VARCHAR(100),
    `data_nascimento` DATE,
    `endereco` TEXT,
    `whatsapp` VARCHAR(50),
    `email` VARCHAR(100),
    `contato_emergencia` VARCHAR(255),
    `profissao` VARCHAR(100),
    `historico_cirurgias` TEXT,
    `doencas_cronicas` TEXT,
    `alergias` TEXT,
    `medicamentos_uso` TEXT,
    `vicios` TEXT,
    `tipo_sanguineo` VARCHAR(10),
    `oscilacao_peso` TEXT,
    `queixa_principal` TEXT,
    `expectativa_resultado` TEXT,
    `saude_mental` TEXT,
    `como_conheceu` VARCHAR(100),
    `autorizacao_imagem` VARCHAR(10),
    `consentimento_lgpd` VARCHAR(10),
    PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
