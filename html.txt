<script>
document.getElementById('formMedicoArkhos').addEventListener('submit', function(e) {
    e.preventDefault();
    const btn = document.getElementById('btnEnviar');
    const status = document.getElementById('statusEnvio');
    
    btn.disabled = true;
    btn.innerText = 'Salvando...';

    const formData = new FormData(this);
    formData.append('action', 'cadastrar_paciente_medico');
    
    // Ajuste dos checkboxes
    ['tabagista', 'etilista', 'sedentario'].forEach(h => {
        formData.set(h, formData.has(h) ? 'Sim' : 'Não');
    });

    fetch('/wp-admin/admin-ajax.php', { method: 'POST', body: formData })
    .then(r => r.json()).then(d => {
        if(d.success) {
            status.innerHTML = '<span style="color:green">✅ Prontuário Salvo!</span>';
            this.reset();
        } else {
            status.innerHTML = '<span style="color:red">❌ Erro ao salvar.</span>';
        }
    }).finally(() => {
        btn.disabled = false;
        btn.innerText = 'Salvar Prontuário';
    });
});
</script>
